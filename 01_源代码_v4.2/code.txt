s = input()
res = []
seen = set()  # 用小写做去重标记
for char in s:
    if char.isalpha():
        lower_c = char.lower()
        if lower_c not in seen:
            seen.add(lower_c)
            res.append(char)
            # 收集满10个提前终止
            if len(res) == 10:
                break
if len(res) >= 10:
    print(''.join(res))
else:
    print("not found")
